CREATE TABLE `perguntas` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `texto_pergunta` text NOT NULL,
 PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4

CREATE TABLE `posts` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `title` varchar(20) NOT NULL,
 `text` varchar(999) NOT NULL,
 `img` varchar(999) NOT NULL,
 `text1` varchar(999) DEFAULT NULL,
 `imgbackground` varchar(999) DEFAULT NULL,
 PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=5 DEFAULT CHARSET=utf8mb4

CREATE TABLE `respostas` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `fk_id_pergunta` int(11) NOT NULL,
 `texto` varchar(255) NOT NULL,
 PRIMARY KEY (`id`),
 KEY `fk_id_pergunta` (`fk_id_pergunta`),
 CONSTRAINT `respostas_ibfk_1` FOREIGN KEY (`fk_id_pergunta`) REFERENCES `perguntas` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4

CREATE TABLE `usuarios` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `email` varchar(80) NOT NULL,
 `senha` varchar(255) NOT NULL,
 `adm` tinyint(1) NOT NULL,
 PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4
